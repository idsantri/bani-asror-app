import { Dialog } from "quasar";
const messageBox = () => {
  Dialog.create({
    title: "Alert",
    message: "Some message"
  })
    .onOk(() => {
      console.log("OK");
    })
    .onCancel(() => {
      console.log("Cancel");
    })
    .onDismiss(() => {
      console.log("I am triggered on both OK and Cancel");
    });
};
export { messageBox };
